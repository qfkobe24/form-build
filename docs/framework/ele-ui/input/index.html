    # 基础用法

    <template>
  <form-build :form-list="formList" ref="form"></form-build>
</template>

<script>
export default {
  data() {
    return {
      formList: [{
        id: 'input_base',
        label: '基础输入框',
        component: 'input',
        props: {
          placeholder: '请输入',
          clearable: true
        }
      }]  
    }
  }
}
</script>

    # 事件和插槽

    <template>
  <div>
    <div style="color:plum">{{tip}}</div>
    <form-build :form-list="formList" ref="form"></form-build>
  </div>
</template>

<script>
export default {
  data() {
    const that = this

    return {
      tip: '',
      formList: [{
        id: 'input_event',
        label: '事件监听',
        component: 'input',
        props: {
          placeholder: '请输入'
        },
        on: {
          change(val) {
            that.tip = `change triggered: ${val}`
          }
        },
        renderSlots(h, val) {
          return [
            h('span', { slot: 'prepend' }, 'https://'),
            h('span', { slot: 'append' }, '.com')
          ]
        }
      }]
    }
  }
}
</script>
