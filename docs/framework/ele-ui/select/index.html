    # 基础用法

    <template>
  <form-build :form-list="formList" ref="form"></form-build>
</template>

<script>
export default {
  data() {
    return {
      formList: [{
        id: 'select',
        label: '基础选择器',
        component: 'select',
        props: {
          placeholder: '请选择',
          multiple: true
        },
        options: [
          { value: 1, label: '红烧鱼', disabled: false },
          { value: 2, label: '酸菜鱼', disabled: false },
          { value: 3, label: '烤鱼', disabled: false }
        ]
      }]
    }
  }
}
</script>

    # 自定义选项模板

    <template>
  <form-build :form-list="formList" ref="form"></form-build>
</template>

<script>
export default {
  data() {
    return {
      formList: [{
        id: 'select',
        label: '自定义选项模板',
        component: 'select',
        props: {
          placeholder: '请选择'
        },
        options: [
          { value: 1, label: '红烧鱼', disabled: false },
          { value: 2, label: '酸菜鱼', disabled: false },
          { value: 3, label: '烤鱼', disabled: false }
        ],
        renderOptionSlot(h, option, index) {
          return h('div', {
            style: {
              display: 'flex',
              justifyContent: 'space-between'
            }
          }, [
            h('span', {}, `${option.label}（招牌菜）`),
            h('span', {}, option.value)
          ])
        }
      }]
    }
  }
}
</script>

    # 可远程搜索

    <template>
  <form-build :form-list="formList" ref="form"></form-build>
</template>

<script>
export default {
  data() {
    const selectSingleOptions = []
    let that = this
    return {
      loading: false,
      formList: [{
        id: 'select',
        label: '远程搜索',
        component: 'select',
        props: {
          placeholder: '请输入关键字搜索',
          filterable: true,
          remote: true,
          loading: that.loading,
          remoteMethod(query) {
            this.loading = true
            setTimeout(() => {
              this.loading = false
              selectSingleOptions.push(
                { value: 1, label: '红烧鱼', disabled: false },
                { value: 2, label: '酸菜鱼', disabled: false },
                { value: 3, label: '烤鱼', disabled: false })
            }, 1000)
          }
        },
        options: selectSingleOptions
      }]
    }
  }
}
</script>

    # Form Item

    名称 类型 默认值 是否必传 说明
    options Array[option对象/option分组对象] true - 选项配置列表
    renderOptionSlot Function(h:createEelement, option: Option对象, index): VNodes - - 自定义选项模板

    # Option对象

    全部支持原文档 (opens new window)

    # Option分组对象

    参数 类型 默认值 是否必传 说明
    label String - 否 分组的组名
    disabled Boolean false 否 是否将该分组下所有选项置为禁用
    options Array[option对象] true 是 选项配置列表
