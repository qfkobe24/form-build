    # Slider 滑块

    # 基础用法

    <template>
  <form-build :form-list="formList" ref="form" labelWidth="120"></form-build>
</template>

<script>
export default {
  data() {
    return {
      formList: [{
        id: 'slider',
        label: '默认',
        component: 'slider',
      }]
    }
  }
}
</script>

    # 自定义初始值

    <template>
  <form-build :form-list="formList" ref="form" :model="model" labelWidth="120"></form-build>
</template>

<script>
export default {
  data() {
    return {
      model: {
        default: 50,
      },
      formList: [{
        id: 'default',
        label: '自定义初始值',
        component: 'slider',
      }]
    }
  },
  methods: {
    formatTooltip(val) {
      return val / 100
    }
  }
}
</script>

    # Tooltip提示

    <template>
  <form-build :form-list="formList" ref="form"  labelWidth="120"></form-build>
</template>

<script>
export default {
  data() {
    return {
      formList: [{
        id: 'hinde',
        label: '隐藏Tooltip',
        component: 'slider',
        props: {
          showTooltip: false
        }
      }, {
        id: 'format',
        label: '格式化Tooltip',
        component: 'slider',
        props: {
          formatTooltip: this.formatTooltip
        }
      }]
    }
  },
  methods: {
    formatTooltip(val) {
      return val / 100
    }
  }
}
</script>

    # 展示标记

    <template>
  <form-build :form-list="formList" ref="form" labelWidth="120"></form-build>
</template>

<script>
export default {
  data() {
    return {
      formList: [{
        id: 'mark',
        label: '展示标识',
        component: 'slider',
        props: {
          marks: {
            0: '0°C',
            8: '8°C',
            37: '37°C',
            50: {
              style: {
                color: '#1989FA'
              },
              label: this.$createElement('strong', '50%')
            }
          }
        },
      }]
    }
  }
}
</script>
