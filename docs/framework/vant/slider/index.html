    # Slider 滑块

    # 基础用法

    <template>
  <form-build :form-list="formList" ref="form" labelWidth="150"></form-build>
</template>

<script>
export default {
  data() {
    return {
      formList: [{
        id: 'slider',
        label: '默认',
        component: 'slider',
      }]
    }
  }
}
</script>

    # 自定义样式

    <template>
  <form-build :form-list="formList" ref="form" labelWidth="150" :model="model"></form-build>
</template>

<script>
export default {
  data() {
    return {
      model: {
        slider: 50
      },
      formList: [{
        id: 'slider',
        label: '自定义样式',
        component: 'slider',
        props: {
          barHeight: '4px',
          activeColor: '#ee0a24'
        }
      }]
    }
  },
  methods: {
    formatTooltip(val) {
      return val / 100
    }
  }
}
</script>

    # 自定义值或范围

    <template>
  <form-build :form-list="formList" ref="form" :model="model" labelWidth="120"></form-build>
</template>

<script>
export default {
  data() {
    return {
      model: {
        default_1: 50,
      },
      formList: [{
        id: 'default_1',
        label: '自定义初始值',
        component: 'slider',
      }, {
        id: 'default_2',
        label: '自定义步长',
        component: 'slider',
        props: {
          step: 10
        },
        on: {
          input: (val) => {
            this.$toast('当前值：' + val)
          } 
        }
      }, {
        id: 'default_3',
        label: '自定义范围',
        component: 'slider',
        props: {
          min: -50,
          max: 50
        },
        on: {
          input: (val) => {
            this.$toast('当前值：' + val)
          } 
        }
      }]
    }
  },
  methods: {
    formatTooltip(val) {
      return val / 100
    }
  }
}
</script>

    # 自定义按钮

    <template>
  <form-build :form-list="formList" ref="form" labelWidth="150"></form-build>
</template>

<script>
export default {
  data() {
    return {
      formList: [{
        id: 'slider',
        component: 'slider',
        label: '自定义按钮',
        renderSlots(h, value) {
          return [h('div', { class: 'custom-slider-button', slot: 'button' }, value)]
        },  
      }]
    }
  }
}
</script>

<style>
.custom-slider-button {
  width: 26px;
  color: rgb(255, 255, 255);
  font-size: 10px;
  line-height: 18px;
  text-align: center;
  background-color: #10a517;
  border-radius: 100px;
}
</style>
